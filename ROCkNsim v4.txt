PNT Trajectory — GPS-denied nav simulation Dependencies: numpy, matplotlib (stdlib only) """ import numpy as np import matplotlib.pyplot as plt import matplotlib.gridspec as gridspec from matplotlib.widgets import Slider, Button, RadioButtons from matplotlib.patches import FancyArrowPatch import warnings warnings.filterwarnings('ignore') # ============================================================================= # ── CLASS-BASED API ────────────────────────────────────────────────────────── # ============================================================================= class OpticalLatticeClock: """ Encapsulates all static clock parameters and core noise/Allan functions. Single source of truth; pass one instance to Servo, Systematics, UKFusion. """ def __init__(self, h0=2.0e-30, hm1=5.0e-17, hm2=1.0e-25, T_int=2.0, dead_time=0.5, nu_clock=429.228e12, N_atoms=1000, contrast=0.85, xi2=1.0, # spin-squeezing factor ξ² ≤ 1 accel_sens=1.0e-14): # PSD coefficients self.h0 = h0 self.hm1 = hm1 self.hm2 = hm2 # Timing self.T = T_int self.Td = dead_time self.Tc = T_int + dead_time # Physical self.nu = nu_clock self.N = N_atoms self.C = contrast self.xi2 = xi2 # ξ² = 1 → standard QPN; ξ² < 1 → squeezed self.accel_sens = accel_sens # Pre-generate noise pool at construction (fix 1) self._pool_size = 2000 self._noise_pool = None self._env_pool = None self._temp_pool = None self.regenerate_noise() # ── noise generation ────────────────────────────────────────────────── def _fft_noise(self, N_pts, dt): """FFT-shaped fractional frequency noise from PSD coefficients.""" f = np.fft.rfftfreq(N_pts, d=dt) f[0] = 1e-9 S = np.zeros_like(f) if self.h0 > 0: S += self.h0 if self.hm1 > 0: S += self.hm1 / f if self.hm2 > 0: S += self.hm2 / f**2 amp = np.sqrt(2.0 * S / dt) * ( np.random.randn(len(f)) + 1j * np.random.randn(len(f))) amp[0] = 0.0 return np.fft.irfft(amp, n=N_pts).real def regenerate_noise(self): """Pre-generate all noise traces once. Called once at startup.""" rng_state = np.random.get_state() np.random.seed(42) # Laser fractional frequency noise (servo cycles) self._noise_pool = self._fft_noise(self._pool_size, self.Tc) # Vibration noise (env tab, fine dt) dt_env = 0.1 n_env = 3000 self._env_pool = ( self._fft_noise(n_env, dt_env) * 1e-4 + self._fft_noise(n_env, dt_env) * 1e-3 ) # Engine harmonics — coherent tones t_env = np.arange(n_env) * dt_env blade_freqs = [25.0, 50.0, 100.0, 200.0] harmonics = sum(0.3e-4 * np.sin(2*np.pi*f*t_env + np.random.uniform(0,2*np.pi)) for f in blade_freqs) self._env_pool_jet = self._env_pool + harmonics self._env_pool_lab = self._fft_noise(n_env, dt_env) * 1e-6 self._t_env = t_env # Temperature fluctuation for dynamic BBR self._temp_pool = 300.0 + 0.5 * self._fft_noise(self._pool_size, self.Tc) np.random.set_state(rng_state) # ── core Allan functions ────────────────────────────────────────────── def allan_free(self, tau): """Analytical Allan deviation from three-term PSD (IEEE 1139).""" s2 = np.zeros_like(tau, dtype=float) if self.h0 > 0: s2 += self.h0 / (2.0 * tau) if self.hm1 > 0: s2 += 2.0 * np.log(2) * self.hm1 if self.hm2 > 0: s2 += (2.0*np.pi**2/3.0) * self.hm2 * tau return np.sqrt(s2) def dick_effect(self, tau): """Dick aliasing penalty from dead time.""" D = self.T / self.Tc dw = np.sqrt(self.h0 * (1-D)**2 / (D * tau)) df = np.sqrt(2.0*np.log(2) * self.hm1 / D) dr = np.sqrt((2*np.pi**2/3) * self.hm2 * tau / D) return np.sqrt(dw**2 + df**2 + dr**2) def dick_interleaved(self, tau): """ Dick effect with dual interleaved ensemble (suggestion 9). Two ensembles interrogated alternately: effective duty cycle → 1. Penalty reduced by factor ~√2 and dead-time aliasing nearly eliminated. Ref: Schioppo et al. 2017 (interleaved OLC). """ D_eff = min(0.98, 2 * self.T / self.Tc) # near-unity effective duty dw = np.sqrt(self.h0 * (1-D_eff)**2 / (D_eff * tau)) df = np.sqrt(2.0*np.log(2) * self.hm1 / D_eff) dr = np.sqrt((2*np.pi**2/3) * self.hm2 * tau / D_eff) return np.sqrt(dw**2 + df**2 + dr**2) def qpn(self): """ Atom projection noise (with spin squeezing). σ_QPN = ξ / (π √N C) where ξ² = xi2 (ξ²<1 → sub-SQL) """ return np.sqrt(self.xi2) / (np.pi * np.sqrt(self.N) * self.C) def allan_total(self, tau, interleaved=False): """Full Allan deviation: laser + Dick + QPN (+ squeezing).""" s2_laser = self.allan_free(tau)**2 dick_fn = self.dick_interleaved if interleaved else self.dick_effect s2_dick = dick_fn(tau)**2 s2_qpn = (self.qpn()**2 * self.T) / tau return np.sqrt(s2_laser + s2_dick + s2_qpn) class Systematics: """ Full 6-term systematic shift model with covariance matrix. Terms: BBR, Lattice, Collision, Zeeman, AC Stark, 2nd-order Doppler. """ # Sr-like coefficients BETA_BBR = -2.13e-4 / 429.228e12 # /K⁴ BETA_LATT = 1.0e-18 # /E_r BETA_COLL = 1.0e-17 # /atom BETA_ZEEMAN = 5.0e-17 # /(µT)² (2nd order Zeeman) BETA_ACST = 3.0e-19 # /(mW/cm²) AC Stark from probe BETA_DOPPLER= 2.0e-20 # /(mm/s)² 2nd order Doppler T_REF = 300.0 # K U_MAGIC = 100.0 # recoil energies # Uncertainty (1-sigma) on each control parameter SIGMA_T = 0.1 # K — temperature control SIGMA_U = 0.5 # E_r — trap depth control SIGMA_N = 50.0 # atoms — number fluctuation SIGMA_B = 0.1 # µT — bias field control SIGMA_I = 0.5 # mW/cm² — probe intensity SIGMA_V = 0.5 # mm/s — residual velocity def __init__(self, T_env=300.1, U_trap=100.5, n_atoms=1000, B_bias=1.0, I_probe=2.0, v_residual=0.3): self.T_env = T_env self.U_trap = U_trap self.n_atoms = n_atoms self.B_bias = B_bias self.I_probe = I_probe self.v_residual = v_residual def shifts(self): """Returns dict of fractional frequency shifts for all 6 terms.""" return { 'BBR': self.BETA_BBR * (self.T_env**4 - self.T_REF**4), 'Lattice': self.BETA_LATT * (self.U_trap - self.U_MAGIC), 'Collision': self.BETA_COLL * self.n_atoms, 'Zeeman': self.BETA_ZEEMAN * self.B_bias**2, 'AC_Stark':self.BETA_ACST * self.I_probe, 'Doppler': self.BETA_DOPPLER* self.v_residual**2, } def uncertainties(self): """Propagated 1-sigma uncertainty for each shift (linear sensitivity).""" return { 'BBR': abs(self.BETA_BBR * 4 * self.T_env**3 * self.SIGMA_T), 'Lattice': abs(self.BETA_LATT * self.SIGMA_U), 'Collision': abs(self.BETA_COLL * self.SIGMA_N), 'Zeeman': abs(self.BETA_ZEEMAN * 2 * self.B_bias * self.SIGMA_B), 'AC_Stark':abs(self.BETA_ACST * self.SIGMA_I), 'Doppler': abs(self.BETA_DOPPLER* 2 * self.v_residual * self.SIGMA_V), } def covariance_matrix(self): """ 6×6 covariance matrix. Off-diagonal: BBR and Lattice are correlated (both T-dependent), BBR and Doppler anti-correlated (thermal motion). All others independent. """ u = self.uncertainties() keys = list(u.keys()) n = len(keys) C = np.diag([u[k]**2 for k in keys]) i_bbr = keys.index('BBR') i_latt = keys.index('Lattice') i_dop = keys.index('Doppler') # BBR–Lattice correlation (both driven by temperature): ρ = 0.4 C[i_bbr, i_latt] = C[i_latt, i_bbr] = 0.4 * u['BBR'] * u['Lattice'] # BBR–Doppler anti-correlation (hot → faster atoms): ρ = -0.3 C[i_bbr, i_dop] = C[i_dop, i_bbr] = -0.3 * u['BBR'] * u['Doppler'] return C, keys def monte_carlo(self, n_samples=20000): """ Monte Carlo accuracy budget. Samples all systematics from correlated Gaussian distribution. Returns array of total fractional frequency errors (shape: n_samples). """ C, keys = self.covariance_matrix() try: L = np.linalg.cholesky(C) except np.linalg.LinAlgError: L = np.linalg.cholesky(C + 1e-60 * np.eye(len(keys))) shifts = np.array([self.shifts()[k] for k in keys]) samples = shifts[:, None] + L @ np.random.randn(len(keys), n_samples) return np.sum(samples, axis=0) def dynamic_bbr(self, T_series): """BBR shift evaluated along a temperature time series.""" return self.BETA_BBR * (T_series**4 - self.T_REF**4) def dynamic_lattice(self, vib_series, coupling=0.01): """Lattice depth fluctuation from vibration (intensity noise).""" return self.BETA_LATT * coupling * vib_series * 1e3 class Servo: """ Discrete-time PI servo with: - VCO/AOM first-order actuator (time constant tau_act) - Atom projection noise (with spin-squeezing) - Closed-loop environmental coupling (vibration → laser freq) - Dynamic systematic offsets (BBR, lattice) fed into error signal """ def __init__(self, clock: OpticalLatticeClock, Kp=0.8, Ki=0.05, tau_act=0.05): self.clock = clock self.Kp = Kp self.Ki = Ki self.tau_act = tau_act def run(self, n_cycles=None, env_coupled=True): """ Run PI servo using pre-generated noise pool. env_coupled: if True, vibration and temperature noise enter the loop. Returns: t, y_free, y_locked, tau_arr, sigma_locked """ clk = self.clock if n_cycles is None: n_cycles = min(clk._pool_size, 800) y_free = clk._noise_pool[:n_cycles].copy() # Environmental coupling (suggestion 5): # vibration → fractional freq via accel_sensitivity # temperature → BBR shift (dynamic) if env_coupled: vib_sub = clk._env_pool_jet[:n_cycles] * clk.accel_sens sys_obj = Systematics() T_series = clk._temp_pool[:n_cycles] bbr_dynamic = sys_obj.dynamic_bbr(T_series) y_free = y_free + vib_sub + bbr_dynamic Tc = clk.Tc alpha = Tc / (Tc + self.tau_act) # actuator discrete lag sigma_qpn = clk.qpn() y_act = np.zeros(n_cycles) y_locked = np.zeros(n_cycles) integral = 0.0 for i in range(n_cycles): y_total = y_free[i] + y_act[i] noise_meas = sigma_qpn * np.random.randn() y_meas = y_total + noise_meas # PI control integral += self.Ki * y_meas * Tc correction = self.Kp * y_meas + integral # Actuator update (first-order lag VCO/AOM model) if i < n_cycles - 1: y_act[i+1] = y_act[i] - alpha * correction y_locked[i] = y_total # Overlapping Allan deviation tau_arr = np.logspace(np.log10(Tc), np.log10(n_cycles * Tc / 10), 80) sigma_arr = np.zeros_like(tau_arr) for idx, tv in enumerate(tau_arr): m = max(1, int(tv / Tc)) avgs = [np.mean(y_locked[k:k+m]) for k in range(0, n_cycles - 2*m, m)] if len(avgs) > 1: sigma_arr[idx] = np.std(avgs) / np.sqrt(2.0) t = np.arange(n_cycles) * Tc return t, y_free, y_locked, tau_arr, sigma_arr class UKFusion: """ Unscented Kalman Filter for GPS + optical clock fusion. State: x = [φ (phase), ω (freq offset), ω̇ (aging/drift rate)] Process: constant aging model φ̈ = ω̇ (3rd-order clock model) Measurements: GPS phase (noisy, with outages), optical phase (low noise) Improvement over v3 scalar KF: - 3-state model captures clock aging (frequency drift rate) - Unscented transform handles mild nonlinearity in phase wrapping - GPS outage simulation: measurement noise → ∞ during outage - Platform acceleration adds to process noise (for mobile platform) """ def __init__(self, clock: OpticalLatticeClock, sigma_gps_ns=5.0, aging_rate=1e-19, accel_noise=1e-14): self.clock = clock self.sigma_gps = sigma_gps_ns * 1e-9 self.aging_rate = aging_rate self.accel_noise = accel_noise def run(self, tau_arr, sigma_optical, gps_outage_frac=0.0): """ Propagate UKF over tau_arr holdover times. gps_outage_frac: fraction of time GPS is denied (0 → always available) Returns optical_err_ns, ukf_err_ns (arrays, same shape as tau_arr) """ Tc = self.clock.Tc dt = Tc # State transition (constant-aging 3-state clock model) F = np.array([[1, dt, 0.5*dt**2], [0, 1, dt ], [0, 0, 1 ]]) # Process noise Q q_phi = (sigma_optical[0] * dt)**2 q_omega = (self.aging_rate * dt)**2 q_aging = (self.aging_rate * dt * 0.01)**2 q_phi += (self.accel_noise * dt)**2 Q = np.diag([q_phi, q_omega, q_aging]) H = np.array([[1, 0, 0]]) R_gps = np.array([[self.sigma_gps**2]]) P = np.eye(3) * 1e-22 x = np.zeros(3) opt_err_ns = [] ukf_err_ns = [] for tau_v in tau_arr: n_steps = max(1, int(tau_v / dt)) # Free optical error (no KF) idx = min(np.searchsorted(tau_arr, tau_v), len(sigma_optical)-1) opt_err_ns.append(sigma_optical[idx] * tau_v * 1e9) # UKF propagation P_k = P.copy() x_k = x.copy() for step in range(n_steps): x_k = F @ x_k P_k = F @ P_k @ F.T + Q is_outage = np.random.rand() < gps_outage_frac if not is_outage: R_eff = R_gps else: R_eff = np.array([[1e6]]) S = H @ P_k @ H.T + R_eff K = P_k @ H.T / S[0, 0] z = self.sigma_gps * np.random.randn() * (1 if not is_outage else 1000) x_k = x_k + K.flatten() * (z - (H @ x_k)[0]) P_k = (np.eye(3) - np.outer(K.flatten(), H)) @ P_k ukf_err_ns.append(np.sqrt(P_k[0, 0]) * 1e9) return np.array(opt_err_ns), np.array(ukf_err_ns) class PNTSimulator: """ Converts clock timing error to navigation position error and simulates a GPS-denied trajectory for a platform. Architecture Notes

Class Hierarchy

OpticalLatticeClock ← physical parameters + noise pool + Allan functions └── Servo ← PI loop with VCO actuator + env coupling └── UKFusion ← 3-state UKF for GPS/optical fusion └── Systematics ← 6-term shift model + covariance + Monte Carlo └── PNTSimulator ← timing → range → dead-reckoning trajectory 

Each class is independent and instantiable from scripts outside the GUI. For example:

clk = OpticalLatticeClock(N_atoms=2000, xi2=0.2) # spin-squeezed clock srv = Servo(clk, Kp=1.2, Ki=0.03) t, yf, yl, tau, sigma = srv.run(n_cycles=5000, env_coupled=True) 

Noise Pre-generation (Fix 1)

The key change from v3: OpticalLatticeClock.__init__() calls regenerate_noise() once, seeded deterministically (np.random.seed(42)). All noise traces — laser, vibration, temperature — are stored as arrays on the clock instance. Slider callbacks read from these cached arrays rather than re-running FFT shaping. This removes the ~50 ms stutter on every slider movement.

Interleaved Dick Suppression (Fix 9)

dick_interleaved() models two atomic ensembles interrogated alternately, offset by T/2. The effective duty cycle approaches 1:

D_eff = min(0.98, 2 * T / Tc) 

For the default parameters (T=2s, Tc=2.5s), single-ensemble D=0.8 → interleaved D_eff=0.98, reducing the Dick-effect white-noise term by ~(1-0.98)²/(1-0.8)² ≈ 100× in variance. The Tab 1 plot shows this as a dotted orange curve sitting close to the free-running laser line.

UKF Clock Model (Fix 6)

The 3-state model [φ, ω, ω̇] adds a drift-rate state absent from the v3 scalar filter. State transition:

F = [[1, dt, dt²/2], [0, 1, dt ], [0, 0, 1 ]] 

The aging term ω̇ captures the slow frequency drift (~10⁻¹⁹ /s²) characteristic of optical clocks. GPS outages are modelled by inflating R by 10⁶ during denied intervals. The 30% outage curve on Tab 4 shows the degraded performance realistically — the filter's phase estimate covariance grows during each outage epoch.

Systematic Covariance Matrix (Fix 3)

The 6×6 matrix encodes two physical correlations:

PairρPhysical reasonBBR – Lattice+0.4Both shift with temperature: hot environment → larger BBR and lattice-depth fluctuation simultaneouslyBBR – Doppler−0.3Hot environment → faster atomic motion → larger 2nd-order Doppler, opposite sign to BBR 

Monte Carlo sampling uses Cholesky decomposition of C to draw correlated Gaussian vectors, giving a physically correct total uncertainty distribution (Tab 6).

PNT Trajectory (Fix 10)

The dead-reckoning model combines two error sources in quadrature:

σ_pos = √(σ_clock² + σ_INS²) 

where σ_clock = c · σ_timing and σ_INS = 10 km · (t/hr)^1.5 (typical medium-grade INS). The (t)^1.5 growth is between the 1.0 (velocity error only) and 2.0 (position-level random walk) limits, representative of a strapdown INS with gyro drift. The crossover between clock-limited and INS-limited regimes is visible on Tab 8 for all three platform types.
