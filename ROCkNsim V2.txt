hm1  = 5.0e-17   # flicker frequency noise floor
hm2  = 1.0e-25   # random-walk frequency noise (very small for good cavity)

# ====================== HELPER FUNCTIONS (First-Principles Core) ======================

def allan_from_psd(tau, h0, hm1, hm2):
    """Exact analytical Allan deviation from PSD coefficients (IEEE standard)."""
    sigma2 = np.zeros_like(tau, dtype=float)
    # White frequency noise (α = 0)
    if h0 > 0:
        sigma2 += h0 / (2.0 * tau)
    # Flicker frequency noise (α = -1)
    if hm1 > 0:
        sigma2 += 2.0 * np.log(2) * hm1
    # Random-walk frequency noise (α = -2)
    if hm2 > 0:
        sigma2 += (2.0 * np.pi**2 / 3.0) * hm2 * tau
    return np.sqrt(sigma2)

def dick_effect(tau, h0, hm1, hm2, T, Tc):
    """Full Dick effect: aliasing penalty from dead time.
    Approximate but rigorous Fourier sum over cycle harmonics (standard in optical-clock literature)."""
    D = T / Tc
    # White-noise contribution (scales as 1/√τ)
    dick_white = np.sqrt(h0 * (1.0 - D)**2 / (D * tau))
    # Flicker contribution (adds to floor, scaled by 1/D)
    dick_flicker = np.sqrt(2.0 * np.log(2) * hm1 / D)
    # Random-walk contribution (minor)
    dick_rw = np.sqrt((2.0 * np.pi**2 / 3.0) * hm2 * tau / D)
    sigma_dick2 = dick_white**2 + dick_flicker**2 + dick_rw**2
    return np.sqrt(sigma_dick2)

def generate_noise_timeseries(N, dt, h0, hm1, hm2):
    """Generate realistic y(t) time series directly from PSD (FFT shaping method).
    Used only for closed-loop servo demonstration (fast and accurate)."""
    t = np.arange(N) * dt
    f = np.fft.rfftfreq(N, d=dt)
    f[0] = 1e-9  # avoid DC singularity
    S_y = np.zeros_like(f)
    if h0 > 0:
        S_y += h0
    if hm1 > 0:
        S_y += hm1 / f
    if hm2 > 0:
        S_y += hm2 / (f**2)
    # Proper normalization for one-sided PSD → time-domain variance
    amp = np.sqrt(2.0 * S_y / dt) * (np.random.randn(len(f)) + 1j * np.random.randn(len(f)))
    amp[0] = 0.0
    y = np.fft.irfft(amp, n=N).real
    return t, y

def simulate_closed_loop_servo(num_cycles=800, T=T_interrogation, Tc=Tc,
                               h0=h0, hm1=hm1, hm2=hm2,
                               Kp=Kp_default, Ki=Ki_default):
    """Closed-loop PI servo simulation.
    Atoms provide error signal every Tc; servo steers laser frequency.
    Returns locked y(t) and its Allan deviation."""
    dt_cycle = Tc
    t, y_free = generate_noise_timeseries(num_cycles, dt_cycle, h0, hm1, hm2)

    y_locked = np.zeros(num_cycles)
    integral = 0.0

    for i in range(num_cycles):
        # Atoms measure detuning during interrogation (free-running noise + current actuator state)
        y_meas = y_free[i] + y_locked[i] if i > 0 else y_free[i]
        # PI control law (negative feedback to null error)
        proportional = Kp * y_meas
        integral += Ki * y_meas * dt_cycle
        correction = proportional + integral
        # Apply correction to actuator for NEXT cycle
        if i < num_cycles - 1:
            y_locked[i + 1] = -correction

    # Compute Allan deviation of the locked clock (numerical estimator)
    tau_values = np.logspace(np.log10(Tc), np.log10(num_cycles * Tc / 10), 80)
    # Simple overlapping Allan (for demo; exact enough)
    sigma_locked = np.zeros_like(tau_values)
    for idx, tau in enumerate(tau_values):
        m = int(tau / dt_cycle)
        if m < 1:
            continue
        y_avg = np.array([np.mean(y_locked[k:k + m]) for k in range(0, num_cycles - 2 * m, m)])
        sigma_locked[idx] = np.std(y_avg) / np.sqrt(2.0)
    return t, y_free, y_locked, tau_values, sigma_locked

fig1, ax1 = plt.subplots(figsize=(9, 5.5))
tau = np.logspace(0, 5, 250)

fig2, ax2 = plt.subplots(figsize=(9, 5.5))

def update_ramsey(T, linewidth_dummy=0):
    # First-principles contrast from white-frequency phase diffusion
    contrast = np.exp(- (2.0 * np.pi**2 * h0 * T**3) / 3.0)
    P = contrast * (np.cos(np.pi * delta_f * T)**2)
    return P, contrast
